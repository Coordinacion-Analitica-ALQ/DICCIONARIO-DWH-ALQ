SELECT FN.PERIODO() AS PERIODO;

INSERT INTO COMERCIAL.PERIODO_DIASH (PERIODO, DIAS_HABILES, TOTAL_DIAS)
		SELECT PERIODO,SUM(PESO_CANATRAD),0 FROM MAESTRO.TABLA_PROYECCION
		WHERE PERIODO = ${PERIODO}
		AND TRIM(CODI_EMPRESA)='GA'
		GROUP BY PERIODO
		ORDER BY PERIODO;

		UPDATE COMERCIAL.PERIODO_DIASH
		SET TOTAL_DIAS=31
		WHERE SUBSTRING(CAST(PERIODO AS STRING),4,2) IN ( '01','03','05','07','08','10','12');

		UPDATE COMERCIAL.PERIODO_DIASH
		SET TOTAL_DIAS=30
		WHERE SUBSTRING(CAST(PERIODO AS STRING),4,2) IN ( '04','06','09','11');

		UPDATE COMERCIAL.PERIODO_DIASH
		SET TOTAL_DIAS=28
		WHERE SUBSTRING(CAST(PERIODO AS STRING),4,2)='02'
