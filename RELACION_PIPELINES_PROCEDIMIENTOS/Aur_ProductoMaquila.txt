select   CONO31 CODI_EMPRESA,RCDK31 CODI_ARTICULO, P02K31  CODIGO_MAQUILA,	PRMD15  NOMBRE_MAQUILA 
FROM   AULT2F3ALQ.INP35 		INNER  join  AULT1F3ALQ.L1P31 E 
ON  CONO35=CONO31 AND PNUM35=RCDK31 		INNER JOIN AULT1F3ALQ.L1P15C  
ON  CONO31=CONO15 AND  P02K31=PSAR15 	
WHERE  CONO35 IN (${VSYS_EMPRESA_MANDATO})
AND PCLS35  IN  (${VSYS_CATEGO_GA}) 	
AND PNUM35 NOT IN ('60101020') AND PRMT15='MAQU'
