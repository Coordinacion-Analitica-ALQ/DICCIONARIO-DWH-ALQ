UPDATE MAESTRO.DIM_CTRO_TRANSPORTE T
  SET NOMBRE_RUTA=R.NOMBRE_RUTA
FROM  MAESTRO.DIM_RUTA  R 
WHERE  R.CODI_EMPRESA=T.CODI_EMPRESA
AND R.CODI_RUTA=T.CODI_RUTA
AND R.CODI_CTROVISTRANS=T.CODI_CTRO_TRANSPORTE;

UPDATE MAESTRO.DIM_CTRO_TRANSPORTE T
  SET NOMBRE_VENDEDOR=R.NOMB_VENDEDOR ,
   CEDULA_VENDEDOR= R.CEDU_VENDEDOR
FROM  MAESTRO.DIM_DATOS_VENDEDOR    R 
WHERE  R.CODI_EMPRESA=T.CODI_EMPRESA
AND R.CODI_VENDEDOR=T.CODIGO_VENDEDOR
AND R.CODI_CTRO_TRANSPORTE=T.CODI_CTRO_TRANSPORTE;
