SELECT FN.PERIODO() AS PERIODO;

UPDATE  COMERCIAL.VENTAS_REPORTE_MENSUAL    
SET CODI_ESTRUCTURA=1
WHERE   PERIODO= ${PERIODO}
AND CODI_EMPRESA IN ('DN','GA');

UPDATE COMERCIAL.VENTAS_REPORTE_MENSUAL
SET CODI_ESTRUCTURA=2
FROM COMERCIAL.VENTAS_REPORTE_MENSUAL AS V INNER JOIN MAESTRO.JERARQUIA_PRODUCTO AS P
ON V.CODI_ARTEMP=P.CODIGO  INNER JOIN MAESTRO.CLIENTES AS C ON V.CODIGO_CLIENTE=C.CODIGO
INNER JOIN MAESTRO.DASA_DISTR_ZONA AS D ON D.CODI_VENDEDOR=C.CODI_VENDEDOR_DASA
AND D.ZONA=C.CODI_ZONA_DASA AND D.DISTRITO=C.CODI_DISTRITO_DASA
WHERE   P.CODI_LINEA in ('C4','D1') AND CODI_REGION='SAB' AND  C.CODI_CANALASOC='TT'
AND C.CODI_EMPRESA IN ('DN','GA')
AND V.PERIODO=${PERIODO};

UPDATE COMERCIAL.VENTAS_REPORTE_MENSUAL
SET CODI_ESTRUCTURA=2
FROM COMERCIAL.VENTAS_REPORTE_MENSUAL AS V INNER JOIN MAESTRO.JERARQUIA_PRODUCTO AS P
ON V.CODI_ARTEMP=P.CODIGO  INNER JOIN MAESTRO.CLIENTES AS C ON V.CODIGO_CLIENTE=C.CODIGO
INNER JOIN MAESTRO.DASA_DISTR_ZONA AS D  ON  D.CODI_VENDEDOR=C.CODI_VENDEDOR_DASA
AND D.ZONA=C.CODI_ZONA_DASA AND D.DISTRITO=C.CODI_DISTRITO_DASA
WHERE   P.CODI_ARTICULO in  ('7006','7007')  and CODI_REGION='SAB' 
AND  C.CODI_CANALASOC='TT'
AND C.CODI_EMPRESA IN ('DN','GA')
and V.PERIODO=${PERIODO};


update COMERCIAL.VENTAS_REPORTE_MENSUAL
SET CODI_ESTRUCTURA=2
FROM COMERCIAL.VENTAS_REPORTE_MENSUAL AS V INNER JOIN MAESTRO.JERARQUIA_PRODUCTO AS P
ON V.CODI_ARTEMP=P.CODIGO  INNER JOIN MAESTRO.CLIENTES AS C ON V.CODIGO_CLIENTE=C.CODIGO
INNER JOIN MAESTRO.DASA_DISTR_ZONA AS D  ON  D.CODI_VENDEDOR=C.CODI_VENDEDOR_DASA
AND D.ZONA=C.CODI_ZONA_DASA AND D.DISTRITO=C.CODI_DISTRITO_DASA
WHERE   P.CODI_LINEA in ('05')  and CODI_REGION='SAB' AND  C.CODI_CANALASOC='TT'
AND C.CODI_EMPRESA IN ('DN','GA') AND P.CODI_ARTICULO LIKE '8%'
and V.PERIODO=${PERIODO};

UPDATE COMERCIAL.VENTAS_REPORTE_MENSUAL
SET CODI_ESTRUCTURA=2
from COMERCIAL.VENTAS_REPORTE_MENSUAL AS V 
INNER JOIN MAESTRO.CLIENTES AS C ON  V.CODIGO_CLIENTE=C.CODIGO
INNER JOIN MAESTRO.JERARQUIA_PRODUCTO AS P ON V.CODI_ARTEMP=P.CODIGO
WHERE SUBSTRING(C.CODI_VENDEDOR_DASA,1,2) = 'PD'
AND C.CODI_UNIDAD in ('RSN','RSS')
AND P.CODI_LINEA IN ('02')
AND P.CODI_GRUPO IN ('08','BS') AND C.CODI_EMPRESA IN ('DN','GA');
