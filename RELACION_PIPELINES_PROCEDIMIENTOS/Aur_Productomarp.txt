Database Properties - Import Query

select   CONO31 CODI_EMPRESA,RCDK31 CODI_ARTICULO, P01K31 CODI_MARC,	PRMD15 NOMBRE_MARC
FROM   AULT1F3ALQ.L1P31 
	INNER JOIN   AULT1F3ALQ.L1P15C 
	 ON  CONO31=CONO15 AND  P01K31=PSAR15
WHERE  CONO31 IN (${VSYS_EMPRESA_MANDATO})
		AND PRMT15='MARC'     
ORDER BY CONO31,RCDK31
