  ,CATEGORIA_COMERCIAL_MODERNO= ''
   WHERE CODI_EMPRESA IN (${VSYS_EMPRESA_MANDATO} )
  and  CODI_LINEA IN (${VSYS_CATEGO_});

update MAESTRO.JERARQUIA_PRODUCTO  P
  SET CATEGORIA_COMERCIAL=  C.CATEGORIA_COMERCIAL
  ,CATEGORIA_COMERCIAL_MODERNO= C.CATEGORIA_COMERCIAL_MODERNO
  FROM  MAESTRO.CATEGORIA_COMERCIAL C
  WHERE  P.CODI_LINEA=C.CODIGO_CATEGORIA 
  AND P.CODI_GRUPO=C.CODIGO_GRUPO
  AND P.CODI_SUBGRUPO=C.CODIGO_SUBGRUPO
  AND P.CODI_TAMANO=C.CODIGO_TAMANO
  AND P.CODI_EMPRESA IN (${VSYS_EMPRESA_MANDATO})
  and P.CODI_LINEA IN (${VSYS_CATEGO_});
