TRUNCATE TABLE COMERCIAL.VENTA_INDICADO_DATOANT;

INSERT INTO COMERCIAL.VENTA_INDICADO_DATOANT
SELECT PERIODO,CODIGO_CLIENTE,CODI_ARTEMP,CODI_CAUSA,NUM_INTERNO ,TIPO_DOCU,CODI_ESTRUCUTRA, 
SUM(LITRO_NETO) AS LITRO,SUM(VALOR_NETO) AS VALOR_NETO ,SUM(LITRO_NC) AS LITRO_NC,
SUM(VALOR_NC) AS VALOR_NC,CODI_SUBCAUSA, SUM( VALOR_FACTURADO ) AS VALOR_FACTURADO, CODIGO_SUBCAUSAL, CODIGO_CAUSAL 
FROM  COMERCIAL.VENTA_INDICADORES_TOTALFK
WHERE   PERIODO IN (   
     CAST(FORMAT_DATE("1%y%m", DATE_SUB(CURRENT_DATE("America/Bogota"), INTERVAL 365 DAY)) AS INT64)
    ,CAST(FORMAT_DATE("1%y%m", DATE_SUB(CURRENT_DATE("America/Bogota"), INTERVAL 395 DAY)) AS INT64)
     ,CAST(FORMAT_DATE("1%y%m", DATE_SUB(CURRENT_DATE("America/Bogota"), INTERVAL 30 DAY)) AS INT64) ) 
GROUP BY PERIODO,CODIGO_CLIENTE,CODI_ARTEMP,CODI_CAUSA,NUM_INTERNO,TIPO_DOCU,CODI_ESTRUCUTRA
 ,CODI_SUBCAUSA,CODIGO_SUBCAUSAL,CODIGO_CAUSAL;
