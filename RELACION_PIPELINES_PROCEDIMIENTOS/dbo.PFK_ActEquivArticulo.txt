SELECT FN.PERIODO() AS PERIODO;

UPDATE COMERCIAL.FK_VENTASA
SET   CODI_EQUIVALENCIA=CODI_ARTICULO
WHERE PERIODO=${PERIODO};


UPDATE COMERCIAL.FK_VENTASA
SET  CODI_ARTICULO=E.CODI_ARTICULO
FROM COMERCIAL.FK_VENTASA AS V INNER JOIN MAESTRO.FK_EQUIVALENCIAS AS E 
ON V.CODI_EQUIVALENCIA=E.CODI_ARTICULO_FK
WHERE  V.PERIODO=${PERIODO};


UPDATE COMERCIAL.FK_VENTASA
SET CODI_ARTICULO='0170000067'
WHERE CODI_ARTICULO='170000067'
AND PERIODO=${PERIODO};


UPDATE COMERCIAL.FK_VENTASA
SET CODI_ARTICULO='0170010188'
WHERE CODI_ARTICULO='170010188'
AND PERIODO=${PERIODO};

UPDATE FK_VENTASA
SET CODI_ARTICULO='0110000383'
WHERE CODI_ARTICULO='110000383'
AND PERIODO=@PERIODO

UPDATE COMERCIAL.FK_VENTASA
SET CODI_ARTICULO='0170010190'
WHERE CODI_ARTICULO='170010190'
AND PERIODO= ${PERIODO};


UPDATE COMERCIAL.FK_VENTASA   
SET CODI_ARTEMP=P.CODIGO
FROM COMERCIAL.FK_VENTASA AS V INNER JOIN MAESTRO.JERARQUIA_PRODUCTO AS P
ON V.CODI_ARTICULO=P.CODI_ARTICULO AND V.CODI_EMPRESA=P.CODI_EMPRESA
WHERE P.CODI_EMPRESA='FK'
and  V.PERIODO= ${PERIODO};

UPDATE COMERCIAL.FK_VENTASA   
SET CODIGO_ARTICULO=P.CODIGO
FROM COMERCIAL.FK_VENTASA AS V INNER JOIN MAESTRO.JERARQUIA_PRODUCTO AS P
ON V.CODI_EQUIVALENCIA=P.CODI_ARTICULO AND V.CODI_EMPRESA=P.CODI_EMPRESA
WHERE P.CODI_EMPRESA='FK'
and  V.PERIODO= ${PERIODO};
