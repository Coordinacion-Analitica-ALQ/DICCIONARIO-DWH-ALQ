SELECT D.CONO90 AS CODI_EMPRESA, D.LOCD90 AS CODI_BODEDESP, D.TELX90 AS  CODI_BODEDEVO, B.STRN20 AS NOMB_BODEDEVO
 FROM AULT2F3ALQ.OEP90 D INNER JOIN
 AULT2F3ALQ.INP20 B
 ON  D.CONO90=B.CONO20 AND D.TELX90=B.STRC20
 WHERE  D.CONO90 IN ('GA','HC','DN','AL','UN','NE','SM','EP','CD','PL','AR','PO','GI', 'NC' ,'AF',  'KZ')
 ORDER BY CONO90
