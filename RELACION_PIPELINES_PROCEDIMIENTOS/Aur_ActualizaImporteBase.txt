SELECT FN.PERIODO() AS PERIODO;

UPDATE VENTAS_REPORTE_EMPRESAS
SET IMPORTE=D.IMPORTE, 
BASE=D.BASE
FROM DATO_IMPORTE AS D
INNER JOIN VENTAS_REPORTE_EMPRESAS AS V ON 
V.CODI_EMPRESA=D.CODI_EMPRESA
AND V.CODI_CLIENTE=D.CODI_CLIENTE 
AND V.CLIE_DIRE=D.SECUENCIA
AND V.NUM_INTERNO=D.NUM_INTERNO
WHERE V.CODI_EMPRESA='EP'
AND V.PERIODO=${PERIODO}
AND D.IMPORTE<>0;
