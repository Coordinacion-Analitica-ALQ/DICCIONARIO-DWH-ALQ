UPDATE MAESTRO.CLIENTES
  SET    CODIGO_REGIONSUPE=''
    ,REGIONAL_SUPERIOR= ''
WHERE CODI_EMPRESA IN ('GA','DN','FK','CD');

UPDATE MAESTRO.CLIENTES C
  SET    CODIGO_REGIONSUPE=R.CODI_REGISUPE, REGIONAL_SUPERIOR= R.REGION_SUPERIOR
FROM  MAESTRO.REGIONAL_SUPERIOR R
WHERE  C.CODI_UNIDAD=R.CODI_UNIDAD and CODI_EMPRESA IN ('GA','DN','FK','CD');

UPDATE MAESTRO.CLIENTES
  SET  CODIGO_SECTOR=CODI_REGION,SECTOR=NOMBRE_REGIONAL
WHERE CODI_EMPRESA IN ('GA','DN','FK','CD');
